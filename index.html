const video = document.getElementById('video'); const canvas = document.getElementById('canvas'); const snapButton = document.getElementById('snap'); const context = canvas.getContext('2d'); // Ask for access to the camera navigator.mediaDevices.getUserMedia({ video: true }) .then((stream) => { video.srcObject = stream; }) .catch((err) => { console.error('Error accessing the camera:', err); }); // Capture frame on button click snapButton.addEventListener('click', () => { canvas.width = video.videoWidth; canvas.height = video.videoHeight; context.drawImage(video, 0, 0, canvas.width, canvas.height); });
